<script>
    const { createApp, ref, computed } = Vue;

    createApp({
        setup() {
            // 控制目前顯示的標籤頁
            const currentTab = ref('itinerary');
            
            // 頂部標題對應
            const tabTitles = {
                itinerary: '我的行程表',
                map: '地點收集與地圖',
                expense: '旅費記帳本',
                shopping: '必買購物清單'
            };

            // 1. 行程表資料 (包含天氣與導航目的地)
            const itinerary = ref([
                { id: 1, time: '09:00', title: '抵達機場', desc: '辦理登機手續', location: '桃園國際機場', temp: '26°C', weatherIcon: 'fas fa-cloud-sun text-yellow-500' },
                { id: 2, time: '13:00', title: '飯店 Check-in', desc: '放行李，稍作休息', location: '新宿車站', temp: '22°C', weatherIcon: 'fas fa-sun text-yellow-500' },
                { id: 3, time: '15:30', title: '熱門景點逛街', desc: '買伴手禮、拍美照', location: '淺草寺', temp: '20°C', weatherIcon: 'fas fa-cloud text-gray-400' }
            ]);

            // 2. 地點連結解析功能
            const linkInput = ref('');
            const savedPlaces = ref([
                {
                    name: '淺草寺',
                    reviews: [
                        { rating: 5, text: '雷門超好拍，附近小吃很多！', link: '' }
                    ]
                }
            ]);

            const parseLink = () => {
                if (!linkInput.value.trim()) return;
                
                // 模擬解析邏輯：隨機抓一個假地點名稱，或者直接叫「新發現的地點」
                // 實務上這裡可以彈出視窗讓使用者確認名稱
                let mockPlaceName = "新發現的地點";
                if(linkInput.value.includes('taipei')) mockPlaceName = "台北101";
                if(linkInput.value.includes('kyoto')) mockPlaceName = "清水寺";

                const placeName = prompt("我們從連結解析出以下地點，請確認名稱 (同名稱會合併評價)：", mockPlaceName);
                
                if (placeName) {
                    const newReview = {
                        rating: Math.floor(Math.random() * 2) + 4, // 模擬 4~5 星
                        text: '從連結自動擷取到的評價內容摘要...',
                        link: linkInput.value
                    };

                    const existingPlace = savedPlaces.value.find(p => p.name === placeName);
                    if (existingPlace) {
                        existingPlace.reviews.push(newReview);
                    } else {
                        savedPlaces.value.push({
                            name: placeName,
                            reviews: [newReview]
                        });
                    }
                    linkInput.value = '';
                }
            };

            // 3. 記帳功能
            const expenses = ref([
                { id: 1, item: '機場快線車票', amount: 350, method: '信用卡' },
                { id: 2, item: '便利商店買水', amount: 45, method: '現金' }
            ]);
            const newExpense = ref({ item: '', amount: '', method: '現金' });

            const totalExpense = computed(() => {
                return expenses.value.reduce((sum, exp) => sum + Number(exp.amount), 0);
            });

            const addExpense = () => {
                if (!newExpense.value.item || !newExpense.value.amount) return;
                expenses.value.unshift({
                    id: Date.now(),
                    item: newExpense.value.item,
                    amount: newExpense.value.amount,
                    method: newExpense.value.method
                });
                newExpense.value.item = '';
                newExpense.value.amount = '';
                newExpense.value.method = '現金'; // 預設回現金
            };

            // 4. 購物清單
            const shoppingList = ref([
                { id: 1, name: '東京香蕉蛋糕 x2', done: false },
                { id: 2, name: '感冒藥、眼藥水', done: true }
            ]);
            const newShoppingItem = ref('');

            const addShoppingItem = () => {
                if (!newShoppingItem.value.trim()) return;
                shoppingList.value.unshift({
                    id: Date.now(),
                    name: newShoppingItem.value,
                    done: false
                });
                newShoppingItem.value = '';
            };

            const removeShoppingItem = (id) => {
                shoppingList.value = shoppingList.value.filter(item => item.id !== id);
            };

            return {
                currentTab,
                tabTitles,
                itinerary,
                linkInput,
                savedPlaces,
                parseLink,
                expenses,
                newExpense,
                totalExpense,
                addExpense,
                shoppingList,
                newShoppingItem,
                addShoppingItem,
                removeShoppingItem
            };
        }
    }).mount('#app');
</script>
