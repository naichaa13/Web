<!-- 引入 Vue 3 和 Firebase (使用 ES Module) -->
<script type="module">
    import { createApp, ref, computed, onMounted } from 'https://unpkg.com/vue@3/dist/vue.esm-browser.js';
    import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
    import { getAuth, signInAnonymously, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
    import { getFirestore, collection, onSnapshot, addDoc, updateDoc, deleteDoc, doc, setDoc } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

    createApp({
        setup() {
            const currentTab = ref('itinerary');
            const tabTitles = { itinerary: '我的行程表', map: '地點收集與地圖', expense: '旅費記帳本', shopping: '共同購物清單' };
            const isMultiplayer = ref(false);

            // 資料庫狀態 (使用 Vue Ref 儲存)
            const itinerary = ref([]);
            const places = ref([]);
            const expenses = ref([]);
            const shoppingList = ref([]);

            // Firebase 變數
            let db, auth;
            // 由於你要放 GitHub，這裡設定了一個機制：如果有環境變數就用，沒有的話就會退回單機版
            // 💡【上傳 GitHub 的提示】：如果你要開啟多人連線，請去 firebase.google.com 申請一個免費專案，並把設定貼在下方
            let firebaseConfig = null;
            try {
                if (typeof __firebase_config !== 'undefined') {
                    firebaseConfig = JSON.parse(__firebase_config);
                }
                /* // 若要在 GitHub 啟用共同編輯，請取消註解並填寫你的 Firebase 資訊：
                firebaseConfig = {
                    apiKey: "YOUR_API_KEY",
                    authDomain: "YOUR_PROJECT.firebaseapp.com",
                    projectId: "YOUR_PROJECT",
                    storageBucket: "YOUR_PROJECT.appspot.com",
                    messagingSenderId: "YOUR_SENDER_ID",
                    appId: "YOUR_APP_ID"
                };
                */
            } catch(e) {}

            const roomId = 'shared-trip-001'; // 所有人共享的房間代號

            onMounted(() => {
                if (firebaseConfig) {
                    initFirebase();
                } else {
                    // 單機測試假資料
                    console.warn("目前為單機模式。若要多人協作請設定 Firebase。");
                    itinerary.value = [{ id: '1', time: '09:00', title: '桃園機場', desc: '辦理登機', location: '桃園國際機場', temp: '26°C', weatherIcon: 'fas fa-sun text-yellow-500' }];
                    places.value = [{ id: '1', name: '淺草寺', reviews: [{ text: '雷門必拍', link: '' }] }];
                }
            });

            // 初始化 Firebase 多人協作
            const initFirebase = async () => {
                const app = initializeApp(firebaseConfig);
                auth = getAuth(app);
                db = getFirestore(app);
                
                await signInAnonymously(auth);
                onAuthStateChanged(auth, (user) => {
                    if (user) {
                        isMultiplayer.value = true;
                        // 監聽四大模組資料
                        setupListener('itinerary', itinerary);
                        setupListener('places', places);
                        setupListener('expenses', expenses);
                        setupListener('shopping', shoppingList);
                    }
                });
            };

            const setupListener = (colName, refObj) => {
                // 使用公共路徑讓所有人都能看到
                const colRef = collection(db, 'artifacts', roomId, 'public', 'data', colName);
                onSnapshot(colRef, (snapshot) => {
                    refObj.value = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
                }, (error) => console.error(error));
            };

            // --- 泛型資料新增/刪除 (支援單機與多人) ---
            const addData = async (colName, dataObj, localRef) => {
                if (isMultiplayer.value) {
                    const colRef = collection(db, 'artifacts', roomId, 'public', 'data', colName);
                    await addDoc(colRef, dataObj);
                } else {
                    localRef.value.unshift({ id: Date.now().toString(), ...dataObj });
                }
            };
            const deleteData = async (colName, id) => {
                if (isMultiplayer.value) {
                    await deleteDoc(doc(db, 'artifacts', roomId, 'public', 'data', colName, id));
                } else {
                    if(colName === 'itinerary') itinerary.value = itinerary.value.filter(i => i.id !== id);
                    if(colName === 'expenses') expenses.value = expenses.value.filter(i => i.id !== id);
                    if(colName === 'shopping') shoppingList.value = shoppingList.value.filter(i => i.id !== id);
                }
            };

            // --- 1. 行程表功能 ---
            const newItinerary = ref({ time: '', title: '', desc: '', location: '', temp: '25°C', weatherIcon: 'fas fa-sun text-yellow-500' });
            const addItinerary = () => {
                if(!newItinerary.value.title) return;
                newItinerary.value.location = newItinerary.value.title; // 預設導航地點同標題
                addData('itinerary', { ...newItinerary.value }, itinerary);
                newItinerary.value.title = '';
            };

            // --- 2. 智慧地點解析 ---
            const linkInput = ref('');
            const isParsing = ref(false);
            const pendingPlace = ref(null);

            const parseLink = () => {
                if (!linkInput.value.trim()) return;
                isParsing.value = true;
                
                // 模擬連結解析邏輯 (正規表達式尋找地名)
                setTimeout(() => {
                    let extractedName = "未命名地點";
                    try {
                        const url = linkInput.value;
                        if (url.includes('place/')) {
                            extractedName = decodeURIComponent(url.split('place/')[1].split('/')[0]).replace(/\+/g, ' ');
                        } else if (url.includes('q=')) {
                            extractedName = decodeURIComponent(url.match(/q=([^&]+)/)[1]).replace(/\+/g, ' ');
                        } else {
                            extractedName = "新景點 (請修改)";
                        }
                    } catch(e) {}
                    
                    pendingPlace.value = {
                        name: extractedName,
                        review: '這看起來是個好地方，先存起來！',
                        link: linkInput.value
                    };
                    isParsing.value = false;
                    linkInput.value = '';
                }, 800);
            };

            const saveParsedPlace = async () => {
                if(!pendingPlace.value.name) return;
                
                const existingPlace = places.value.find(p => p.name === pendingPlace.value.name);
                const newReview = { text: pendingPlace.value.review, link: pendingPlace.value.link };

                if (isMultiplayer.value) {
                    if (existingPlace) {
                        const placeDoc = doc(db, 'artifacts', roomId, 'public', 'data', 'places', existingPlace.id);
                        await updateDoc(placeDoc, { reviews: [...existingPlace.reviews, newReview] });
                    } else {
                        await addData('places', { name: pendingPlace.value.name, reviews: [newReview] }, places);
                    }
                } else {
                    if (existingPlace) {
                        existingPlace.reviews.push(newReview);
                    } else {
                        places.value.unshift({ id: Date.now().toString(), name: pendingPlace.value.name, reviews: [newReview] });
                    }
                }
                pendingPlace.value = null;
            };

            // --- 3. 記帳功能 ---
            const newExpense = ref({ item: '', amount: '', method: '現金' });
            const expensesAmount = computed(() => expenses.value.map(e => Number(e.amount)));
            const totalExpense = computed(() => expensesAmount.value.reduce((a, b) => a + b, 0));

            const addExpense = () => {
                if (!newExpense.value.item || !newExpense.value.amount) return;
                addData('expenses', { ...newExpense.value }, expenses);
                newExpense.value.item = ''; newExpense.value.amount = '';
            };

            // --- 4. 購物清單 ---
            const newShoppingItem = ref('');
            const addShoppingItem = () => {
                if (!newShoppingItem.value.trim()) return;
                addData('shopping', { name: newShoppingItem.value, done: false }, shoppingList);
                newShoppingItem.value = '';
            };
            const toggleShoppingDone = async (item) => {
                if (isMultiplayer.value) {
                    const itemDoc = doc(db, 'artifacts', roomId, 'public', 'data', 'shopping', item.id);
                    await updateDoc(itemDoc, { done: !item.done });
                } else {
                    item.done = !item.done;
                }
            };

            return {
                currentTab, tabTitles, isMultiplayer,
                itinerary, newItinerary, addItinerary, deleteData,
                linkInput, isParsing, pendingPlace, parseLink, saveParsedPlace, places,
                expenses, newExpense, totalExpense, addExpense,
                shoppingList, newShoppingItem, addShoppingItem, toggleShoppingDone
            };
        }
    }).mount('#app-container');
</script>
</body>
</html>
